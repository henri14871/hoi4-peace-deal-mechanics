on_actions = {
	
	on_weekly = {
		effect = {
			if = {
				limit = {
					has_country_flag = pd_ai_want_peace
					var:pd_ai_want_peace_with = {
						NOT = { pd_peace_deal_can_initiate = yes }
					}
				}
				pd_ai_clear_peace_intent = yes
			}
			if = {
				limit = {
					is_ai = yes
					OR = {
						has_war = yes
						has_country_flag = ceasefire_active
					}
					pd_rule_ai_can_initiate = yes
					NOT = { has_country_flag = pd_waiting_response }
				}
				set_temp_variable = { best_target_score = -1 }
				every_country = {
					limit = {
						OR = {
							has_war_with = ROOT
							pd_has_active_ceasefire_with_root = yes
						}
						pd_peace_deal_can_initiate = yes
					}
					pd_ai_calculate_peace_scores = yes
					if = {
						limit = {
							check_variable = { ai_target_priority > -1 }
						}
						set_temp_variable = { candidate_keep = 0 }
						if = {
							limit = {
								ROOT = {
									has_country_flag = pd_ai_want_peace
									check_variable = { pd_ai_want_peace_with = PREV }
								}
							}
							set_temp_variable = { candidate_keep = 1 }
						}

						set_temp_variable = { candidate_vp_diff = ROOT.vp@THIS }
						subtract_from_temp_variable = { candidate_vp_diff = THIS.vp@ROOT }
						round_temp_variable = candidate_vp_diff
						clamp_temp_variable = {
							var = candidate_vp_diff
							min = 0
							max = 999
						}

						set_temp_variable = { candidate_total = ai_target_priority }
						multiply_temp_variable = { candidate_total = 10000000 }

						if = {
							limit = {
								check_variable = { candidate_keep = 1 }
							}
							add_to_temp_variable = { candidate_total = 100000 }
						}

						set_temp_variable = { candidate_vp_weight = candidate_vp_diff }
						multiply_temp_variable = { candidate_vp_weight = 100 }
						add_to_temp_variable = { candidate_total = candidate_vp_weight }

						if = {
							limit = {
								is_in_faction = no
							}
							add_to_temp_variable = { candidate_total = 10 }
						}

						if = {
							limit = {
								is_major = no
							}
							add_to_temp_variable = { candidate_total = 1 }
						}

						if = {
							limit = {
								ROOT = {
									check_variable = { best_target_score < PREV.candidate_total }
								}
							}
							ROOT = {
								set_temp_variable = { best_target_score = PREV.candidate_total }
								set_temp_variable = { best_target = PREV }
							}
						}
					}
				}
				if = {
					limit = {
						check_variable = { best_target_score > -1 }
					}
					var:best_target = {
						ROOT = {
							set_country_flag = pd_ai_want_peace
							set_variable = { pd_ai_want_peace_with = PREV }
						}
					}
				}
				else = {
					pd_ai_clear_peace_intent = yes
				}
			}
		}
	}
	
	# ROOT is capitulated country, FROM is winner
	on_capitulation = {
		effect = {
			if = {
				limit = {
					is_in_faction = yes
				}
				random_country = {
					limit = {
						is_in_faction_with = ROOT
						is_faction_leader = yes
						has_war_with = FROM
					}
					set_temp_variable = { pts = FROM.vp@ROOT }
					divide_temp_variable = { pts = 2 }
					round_temp_variable = pts
					add_to_variable = { FROM.vp@THIS = pts }
				}
			}
			set_temp_variable = { tot_pts = 0 }
			every_owned_state = {
				limit = {
					is_controlled_by = FROM
				}
				add_to_temp_variable = { tot_pts = THIS.state_strategic_value }
			}
			divide_temp_variable = { tot_pts = 2 }
			round_temp_variable = tot_pts
			add_to_variable = { FROM.vp@ROOT = tot_pts }
		}
	}

	on_war_relation_added = {
		effect = {
			set_variable = { vp@FROM = 0 }
			set_variable = { FROM.vp@ROOT = 0 }
		}
	}
	
	on_nuke_drop = {
		effect = {
			add_to_variable = { vp@FROM.owner = 500 }
		}
	}

	on_state_control_changed = {
		effect = {
			set_temp_variable = { state_vp = FROM.FROM.state_strategic_value }
			set_temp_variable = { tot_pts = 0 }
			every_country = {
				limit = {
					has_war_with = FROM
				}
				add_to_temp_variable = { tot_pts = num_armies_in_state@FROM.FROM }
			}
			every_country = {
				limit = {
					has_war_with = FROM
				}
				set_temp_variable = { pts = num_armies_in_state@FROM.FROM }
				divide_temp_variable = { pts = tot_pts }
				multiply_temp_variable = { pts = state_vp }
				add_to_variable = { vp@FROM = pts }
				subtract_from_variable = { FROM.vp@THIS = pts }
				clamp_variable = {
					var = FROM.vp@THIS
					min = 0
				}
			}
		}
	}
}
