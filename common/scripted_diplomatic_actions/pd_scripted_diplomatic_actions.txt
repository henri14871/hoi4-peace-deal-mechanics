scripted_diplomatic_actions = {

	offer_peace_deal = {
	
		# root is the initiator of action and this is the target country
		allowed = {
			always = yes
		}
		visible = {
			pd_rule_ai_participates = yes
			pd_has_peace_deal_relation = yes
		}
		selectable = { 
			pd_peace_deal_can_initiate = yes
		}

		requires_acceptance = yes
		show_acceptance_on_action_button = yes # set to no if acceptance shouldn't be displayed on the action button
		cost = 0

		icon = 1

		on_sent_effect = {
			ROOT = { set_country_flag = pd_waiting_response }
		}

		complete_effect = {
			clear_vp = yes
			execute_peace_deal = yes
			ROOT = { 
				clr_country_flag = pd_waiting_response
				clr_country_flag = pd_ai_want_peace
				clear_variable = pd_ai_want_peace_with 
			}
		}
		
		reject_effect = {
			ROOT = { 
				clr_country_flag = pd_waiting_response
				clr_country_flag = pd_ai_want_peace
				clear_variable = pd_ai_want_peace_with 
			}
		}
		
		send_scripted_gui = pd_offer_peace_deal
		reset_send_effect = {
			set_variable = { ROOT.pd_deal_target = THIS }
			if = {
				limit = {
					ROOT = {
						is_ai = yes
						has_country_flag = pd_ai_want_peace
						check_variable = { ROOT.pd_ai_want_peace_with = THIS }
					}
				}
				pd_ai_prepare_peace_deal = yes
			}
			else = {
				ROOT = { clr_country_flag = pd_root_gives }
				reset_peace_deal_proposal = yes
			}
			pd_compute_cached_term_score = yes
		}
		
		can_be_sent = { # send button will be disabled if no
			pd_rule_ai_participates = yes
			pd_peace_deal_can_initiate = yes
			pd_peace_deal_proposal_is_legal = yes
			pd_peace_deal_has_terms = yes
			OR = {
				ROOT = { is_ai = no }
				pd_ai_current_proposal_is_valid = yes
			}
		}

		can_be_accepted = {
			pd_rule_ai_participates = yes
			pd_peace_deal_participants_valid = yes
			pd_peace_deal_proposal_is_legal = yes
			pd_peace_deal_has_terms = yes
		}
		
		receive_scripted_gui = pd_offer_peace_deal_receive

		ai_desire = {
			base = -10000
			modifier = {
				add = 9900
				pd_rule_ai_can_initiate = yes
				ROOT = { has_country_flag = pd_ai_want_peace }
				check_variable = { ROOT.pd_ai_want_peace_with = THIS }
			}
		}
		
		ai_acceptance = {
			pd_war_pressure = {
				base = 0
				modifier = {
					set_temp_variable = { pressure = 0 }

					set_temp_variable = { sp = var:THIS.surrender_progress }
					multiply_temp_variable = { sp = 220 }
					round_temp_variable = sp
					subtract_from_temp_variable = { sp = 40 }
					clamp_temp_variable = {
						var = sp
						min = 0
						max = 180
					}
					add_to_temp_variable = { pressure = sp }

					set_temp_variable = { ws = var:THIS.has_war_support }
					multiply_temp_variable = { ws = -120 }
					round_temp_variable = ws
					add_to_temp_variable = { ws = 60 }
					clamp_temp_variable = {
						var = ws
						min = -60
						max = 60
					}
					add_to_temp_variable = { pressure = ws }

					set_temp_variable = { s = var:THIS.has_stability }
					multiply_temp_variable = { s = -120 }
					round_temp_variable = s
					add_to_temp_variable = { s = 60 }
					clamp_temp_variable = {
						var = s
						min = -60
						max = 60
					}
					add_to_temp_variable = { pressure = s }

					set_temp_variable = { rvp = ROOT.vp@THIS }
					subtract_from_temp_variable = { rvp = THIS.vp@ROOT }
					round_temp_variable = rvp
					clamp_temp_variable = {
						var = rvp
						min = -160
						max = 160
					}
					add_to_temp_variable = { pressure = rvp }

					set_temp_variable = { str = THIS.enemies_strength_ratio@ROOT }
					multiply_temp_variable = { str = 140 }
					round_temp_variable = str
					subtract_from_temp_variable = { str = 140 }
					clamp_temp_variable = {
						var = str
						min = -120
						max = 160
					}
					add_to_temp_variable = { pressure = str }

					set_temp_variable = { wt = global.threat }
					multiply_temp_variable = { wt = -50 }
					round_temp_variable = wt
					clamp_temp_variable = {
						var = wt
						min = -50
						max = 0
					}
					add_to_temp_variable = { pressure = wt }

					add = var:pressure
				}
			}
			
			pd_peace_terms = {
				base = 0
				modifier = {
					add = ROOT.pd_cached_term_score
				}
			}

			pd_proposal_cost_soft = {
				base = 0
				modifier = {
					set_temp_variable = { tc = ROOT.pd_total_cost }
					divide_temp_variable = { tc = 5 }
					round_temp_variable = tc
					clamp_temp_variable = {
						var = tc
						min = -20
						max = 40
					}
					multiply_temp_variable = { tc = -1 }
					add = var:tc
				}
			}
			
			pd_geopolitical_situation = {
				base = 0
				modifier = {
					add = -25
					is_major = yes
				}
				modifier = {
					add = -15
					is_in_faction = yes
				}
			}

			pd_acceptance_stance = {
				base = 0
				modifier = {
					add = -25
					has_game_rule = {
						rule = pd_ai_acceptance_stance
						option = PD_AI_HARDLINE
					}
				}
				modifier = {
					add = 25
					has_game_rule = {
						rule = pd_ai_acceptance_stance
						option = PD_AI_CONCILIATORY
					}
				}
			}
		}
	}
}
